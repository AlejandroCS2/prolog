# Cálculos de 10 áreas en CLISP

```lsp
(defun cuadrado()
    (princ "Dame el lado: ")
    (setq num1 (read))
    (setq resultado (expt num1 2 ))
    (write resultado)
)
(defun rectangulo()
    (princ "Dame la base: ")
    (setq num1 (read))
    (princ "Dame la altura: ")
    (setq num2 (read))
    (setq resultado (* num1 num2 ))
    (write resultado)
)
(defun triangulo()
    (princ "Dame la base: ")
    (setq num1 (read))
    (princ "Dame la altura: ")
    (setq num2 (read))
    (setq resultado (/ ( * num1 num2 ) 2 ))
    (write resultado)
)
(defun circulo()
    (princ "Dame el radio: ")
    (setq num1 (read))
    (setq resultado (* pi (* num1 num1 )))
    (write resultado)
)
(defun trapecio()
    (princ "Dame la base 1: ")
    (setq num1 (read))
    (princ "Dame la base 2: ")
    (setq num2 (read))
    (princ "Dame la altura: ")
    (setq num3 (read))
    (setq resultado (/ (* (+ num1 num2) num3) 2))
    (write resultado)
)
(defun rombo()
    (princ "Dame la diagonal 1: ")
    (setq num1 (read))
    (princ "Dame la diagonal 2: ")
    (setq num2 (read))
    (setq resultado (* num1 num2 0.5))
    (write resultado)
)
```
