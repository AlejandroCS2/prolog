# Cálculos de 10 áreas en CLISP

```lsp
(defun cuadrado()
    (princ "Dame el lado: ")
    (setq num1 (read))
    (setq resultado (expt num1 2 ))
    (write resultado)
)
(defun rectangulo()
    (princ "Dame la base: ")
    (setq num1 (read))
    (princ "Dame la altura: ")
    (setq num2 (read))
    (setq resultado (* num1 num2 ))
    (write resultado)
)
(defun triangulo()
    (princ "Dame la base: ")
    (setq num1 (read))
    (princ "Dame la altura: ")
    (setq num2 (read))
    (setq resultado (/ ( * num1 num2 ) 2 ))
    (write resultado)
)
(defun circulo()
    (princ "Dame el radio: ")
    (setq num1 (read))
    (setq resultado (* pi (* num1 num1 )))
    (write resultado)
)
(defun trapecio()
    (princ "Dame la base 1: ")
    (setq num1 (read))
    (princ "Dame la base 2: ")
    (setq num2 (read))
    (princ "Dame la altura: ")
    (setq num3 (read))
    (setq resultado (/ (* (+ num1 num2) num3) 2))
    (write resultado)
)
(defun rombo()
    (princ "Dame la diagonal 1: ")
    (setq num1 (read))
    (princ "Dame la diagonal 2: ")
    (setq num2 (read))
    (setq resultado (* num1 num2 0.5))
    (write resultado)
)
(defun pentagono()
    (princ "Dame el lado del pentágono: ")
    (setq lado (read))
    (princ "Dame la apotema del pentágono: ")
    (setq apotema (read))
    (setq resultado (/ (* 5 lado apotema) 2))
    (write resultado)
)
(defun hexagono()
    (princ "Dame el lado del hexágono: ")
    (setq lado (read))
    (princ "Dame la apotema del hexágono: ")
    (setq apotema (read))
    (setq resultado (* 3 (* 6 lado apotema)))
    (write resultado)
)
(defun octagono()
    (princ "Dame el lado del octágono: ")
    (setq lado (read))
    (princ "Dame la apotema del octágono: ")
    (setq apotema (read))
    (setq resultado (* 4 (* 8 lado apotema)))
    (write resultado)
)
(defun romboide()
    (princ "Dame la base del romboide: ")
    (setq base (read))
    (princ "Dame la altura del romboide: ")
    (setq altura (read))
    (setq resultado (* base altura))
    (write resultado)
)
```
